import numpy as np
from sklearn.linear_model import LinearRegression

a = input().split()
b = input().split()
x = int(input())

m = list(map(float, a))
n = list(map(float, b))

X = np.array(m).reshape(-1,1)
y = np.array(n).reshape(-1,1)

linear_model = LinearRegression()
linear_model.fit(X,y)

predict = linear_model.predict(np.array([[x]]))[0][0]

print(f'Predict 12 inch cost:${predict:.2f}')
